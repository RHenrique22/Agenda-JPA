CREATE TABLE REUNIAO (ID  SERIAL NOT NULL, ASSUNTO VARCHAR(255), DATAHORA TIMESTAMP, PRIMARY KEY (ID))
CREATE TABLE CONVIDADO (NOME VARCHAR(255) NOT NULL, EMAIL VARCHAR(255), EMPRESA VARCHAR(255), PRIMARY KEY (NOME))
CREATE TABLE PARTICIPANTE (NOME VARCHAR(255) NOT NULL, EMAIL VARCHAR(255), PRIMARY KEY (NOME))
CREATE TABLE REUNIAO_PARTICIPANTE (Reuniao_ID INTEGER NOT NULL, participantes_NOME VARCHAR(255) NOT NULL, PRIMARY KEY (Reuniao_ID, participantes_NOME))
CREATE TABLE CONVIDADO_REUNIAO (Convidado_NOME VARCHAR(255) NOT NULL, reunioes_ID INTEGER NOT NULL, PRIMARY KEY (Convidado_NOME, reunioes_ID))
CREATE TABLE PARTICIPANTE_REUNIAO (Participante_NOME VARCHAR(255) NOT NULL, reunioes_ID INTEGER NOT NULL, PRIMARY KEY (Participante_NOME, reunioes_ID))
